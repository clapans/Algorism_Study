## 키패드 누르기



```python
numbers = [7, 0, 8, 2, 8, 3, 1, 5, 7, 6, 2]
hand = 'left'

res = ''
position = {'left' : [0,-1], 'right' : [0,1], # key에 해당하는 좌표를 딕셔너리로 표현
                1 : [3,-1], 2 : [3,0], 3 : [3,1],
                4 : [2,-1], 5 : [2,0], 6 : [2,1],
                7 : [1,-1], 8 : [1,0], 9 : [1,1],
                            0 : [0,0]
            }

for i in numbers:
    tmp = i % 3	# % 3 으로 [1,4,7],[2,5,8],[3,6,9]를 묶음으로 구분가능 
    if i == 0 or tmp == 2:
        distance = []
        for t in ['left','right']:	#left,right에서 동일한 연산을 하기에 for문을 이용
            x = abs(position[t][0] - position[i][0])
            y = abs(position[t][1] - position[i][1])
            distance.append(x+y)
        if distance[0] < distance[1]:
            res += 'L'
            position['left'] = position[i]
        elif distance[0] == distance[1]:
            if hand == 'left':
                res += 'L'
                position['left'] = position[i]
            else:
                res += 'R'
                position['right'] = position[i]
        else:
            res += 'R'
            position['right'] = position[i]
    elif tmp == 1:
        res += 'L'
        position['left'] = position[i]
    else:
        res += 'R'
        position['right'] = position[i]

print(res)
```
